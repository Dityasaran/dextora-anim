## Usage

Every animation name is passed to the `data-anim` attribute:

<CodeBlock
  code={'<div\n  data-anim="fadeInUp"\n  data-anim-duration="800ms"\n>\n  This element fades in from below over 800ms\n</div>'}
  lang="html"
/>

See [Attributes](/data-anim/docs/attributes) for all customization options, or try them in the [Playground](/data-anim/playground).

<script>{`
  var inPlaceAnims = ['bounce', 'shake', 'pulse', 'wobble', 'flip', 'swing', 'rubberBand'];

  function playBox(box, name) {
    box.style.animation = 'none';
    box.style.opacity = '';
    box.style.transform = '';
    box.style.filter = '';
    box.style.clipPath = '';
    box.style.maxWidth = '';
    box.style.overflow = '';
    box.style.whiteSpace = '';
    box.offsetHeight;

    if (inPlaceAnims.indexOf(name) === -1) {
      if (name === 'clipReveal') {
        box.style.clipPath = 'inset(0 100% 0 0)';
      } else if (name === 'typewriter') {
        box.style.maxWidth = '0';
        box.style.overflow = 'hidden';
        box.style.whiteSpace = 'nowrap';
      } else {
        box.style.opacity = '0';
      }
    }
    box.offsetHeight;
    box.style.animation = 'da-' + name + ' 800ms ease both';
  }

  function resetBox(box) {
    box.style.animation = 'none';
    box.style.opacity = '';
    box.style.transform = '';
    box.style.filter = '';
    box.style.clipPath = '';
    box.style.maxWidth = '';
    box.style.overflow = '';
    box.style.whiteSpace = '';
  }

  document.querySelectorAll('.demo-card').forEach(function(card) {
    var name = card.getAttribute('data-demo-anim');
    var box = card.querySelector('[data-demo-box]');
    if (!box || !name) return;

    card.addEventListener('mouseenter', function() {
      playBox(box, name);
    });
    card.addEventListener('mouseleave', function() {
      resetBox(box);
    });
    card.addEventListener('click', function() {
      playBox(box, name);
    });
  });

  document.getElementById('replay-all').addEventListener('click', function() {
    document.querySelectorAll('[data-demo-box]').forEach(function(box) {
      var name = box.getAttribute('data-demo-box');
      if (name) playBox(box, name);
    });
  });

  document.querySelectorAll('.replay-group-btn').forEach(function(btn) {
    btn.addEventListener('click', function() {
      var section = btn.closest('.demo-section');
      if (!section) return;
      section.querySelectorAll('[data-demo-box]').forEach(function(box) {
        var name = box.getAttribute('data-demo-box');
        if (name) playBox(box, name);
      });
    });
  });
`}</script>
